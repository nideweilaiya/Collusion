import { useState, useEffect } from 'react';

interface UserData {
  id: number;
  name: string;
  email: string;
  role: string;
  lastLogin: Date;
  permissions: string[];
}

function useUserFilter(users: UserData[], searchTerm: string) {
  const [filteredUsers, setFilteredUsers] = useState<UserData[]>([]);

  useEffect(() => {
    const filtered = users
      .filter(u => u.name.toLowerCase().includes(searchTerm.toLowerCase()))
      .sort((a, b) => a.name.localeCompare(b.name));
    setFilteredUsers(filtered);
  }, [users, searchTerm]);

  return {
    filteredUsers,
    adminCount: filteredUsers.filter(u => u.role === 'admin').length,
    recentLogins: filteredUsers.filter(u => {
      const daysSinceLogin = (Date.now() - u.lastLogin.getTime()) / (1000 * 60 * 60 * 24);
      return daysSinceLogin < 7;
    })
  };
}
